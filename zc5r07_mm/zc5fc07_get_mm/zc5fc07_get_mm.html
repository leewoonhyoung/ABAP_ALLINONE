FUNCTION ZC5FC07_GET_MM .
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(PI_WERKS) TYPE ZC5T07_MM-WERKS
*" EXPORTING
*" REFERENCE(PE_CODE) TYPE CHAR1
*" REFERENCE(PE_MSG) TYPE CHAR200
*" TABLES
*" PT_MM STRUCTURE ZC5T07_MM
*" PT_SEL_OPT1
*" PT_SEL_OPT2
*"----------------------------------------------------------------------
* Global data declarations

  CLEAR   : pt_mm.
  REFRESH : pt_mm.

  SELECT matnr werks stlan stlnr
         stlal losvn losbs bmein
    INTO CORRESPONDING FIELDS OF TABLE pt_mm
    FROM zc5T07_mm
    WHERE werks =  pi_werks
      AND matnr IN pt_sel_opt1
      AND stlan IN pt_sel_opt2.

  IF sy-subrc NE 0.
    pe_code = 'E'.
    pe_msg  = 'MM Data not found'.
  ELSE.
    pe_code = 'S'.
  ENDIF.


ENDFUNCTION.
